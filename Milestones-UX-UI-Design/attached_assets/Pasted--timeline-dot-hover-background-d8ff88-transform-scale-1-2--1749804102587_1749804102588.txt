        let currentStartIndex = 5; // Start showing from index 5 (last 3 items)
        const itemsToShow = 3;

        function renderTimeline() {
            const timeline = document.getElementById('timeline');
            const scrollBtn = document.getElementById('scrollBtn');
            
            // Clear timeline
            timeline.innerHTML = '';
            
            // Show current 3 items
            for (let i = currentStartIndex; i < Math.min(currentStartIndex + itemsToShow, timelineData.length); i++) {
                const item = timelineData[i];
                const timelineItem = createTimelineItem(item);
                timeline.appendChild(timelineItem);
                
                // Add visible class with delay for animation
                setTimeout(() => {
                    timelineItem.classList.add('visible');
                }, (i - currentStartIndex) * 100);
            }
            
            // Show/hide scroll button
            if (currentStartIndex > 0) {
                scrollBtn.classList.add('show');
            } else {
                scrollBtn.classList.remove('show');
            }
        }

        function createTimelineItem(data) {
            const item = document.createElement('div');
            item.className = 'timeline-item';
            item.innerHTML = `
                <div class="timeline-dot"></div>
                <div class="timeline-year">${data.year}</div>
                <div class="timeline-content">
                    <div class="timeline-date">${data.date}</div>
                    <div class="timeline-title-item">${data.title}</div>
                    <div class="timeline-description">${data.description}</div>
                </div>
            `;
            return item;
        }

        function loadPastItems() {
            if (currentStartIndex > 0) {
                currentStartIndex = Math.max(0, currentStartIndex - 1);
                renderTimeline();
            }
        }

        // Initialize timeline
        window.addEventListener('load', function() {
            renderTimeline();
        });
